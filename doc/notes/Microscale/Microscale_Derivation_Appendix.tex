Dl^2(\textbf{x}) &=& 2 (\textbf{x}^b - \textbf{x}^a) \cdot (\textbf{u}^b - \textbf{u}^a) \nonumber\\
&=& l(\textbf{x})\textbf{n} \cdot (\textbf{u}^b - \textbf{u}^a) 
\label{Dl^2}
\end{eqnarray}

\end{equation}
